(function() {
var loca = window.location;
var host = loca.host;
var path = loca.pathname;
var body = get('body');
var text = '1000000000'

switch(path) {
    case 'detail.m.tmall.com':
        tmallFn();
        break;
    case '/item.htm':
        tmallFn();
        break;
    case 'h5.m.taobao.com':
        taobaoFn();
        break;
    case '/awp/core/detail.htm':
        taobaoFn();
        break;
    case '/m/jusp/alone/detailwap/mtp.htm':
        jhsFn();
        break;
    default:
        break;
}

function taobaoFn() {
    var elem = body.children[0];
    var len = elem.children.length;
    if (elem && elem.nodeName === 'DIV' && len == 1) {
        body.removeChild(elem);
        body.style.paddingTop = 0;
    }
    var oBar = get('.bottom-bar');
    if (oBar && !oBar.length) {
        deleteBar(oBar, text);
    }
}

function jhsFn() {
    var top = get('#J_topbar');
    body.removeChild(top);
    var oBottom = get('#J_bottombar');
    var oBar = oBottom.children[0];
    if (oBar) {
        deleteBar(oBar, text);
    }
}

function tmallFn() {
    var tmElem = get("#J_smartloader");
    var page = get(".page");
    var header_bg = page.children[0]
    if (tmElem && header_bg) {
        var tmParent = tmElem.parentNode;
        tmParent.removeChild(tmElem);
        page.removeChild(header_bg);
        body.style.paddingTop = 0;
    }
    var oBar = document.querySelector('#s-actionBar-container');
    if (oBar) {
        deleteBar(oBar, text);
    }
}
function get(selector) {
    var elem = document.querySelectorAll(selector);
    if (elem.length === 1) {
        return elem[0];
    } else {
        return elem;
    }
}
function deleteBar(oBar, context) {
    var newElem = createElem('div', context, {
        background: '#fff',
        position: 'absolute',
        width: '100%',
        height: '110%',
        top: '-5%',
        left: '0',
        zIndex: '99999999999',
        borderTop: '1px solid #fff',
        textAlign: 'center',
        overflow: 'hidden',
        lineHeight: oBar.offsetHeight + 'px',
        fontSize: '16px',
    });
    oBar.appendChild(newElem);
}
function createElem(elemName, context, styles) {
    elemName = elemName || 'div';
    context = context || '';
    var oElem = document.createElement(elemName);
    oElem.innerHTML = context;
    if (styles) {
        for (k in styles) {
            oElem['style'][k] = styles[k];
        }
    }
    return oElem;
}

})()
